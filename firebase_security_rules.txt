rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isParentOfDevice(deviceId) {
      return isAuthenticated() && 
             get(/databases/$(database)/documents/devices/$(deviceId)).data.parentId == request.auth.uid;
    }

    function isDeviceItself(deviceId) {
      // Assuming devices authenticate with a custom token where uid == deviceId
      return isAuthenticated() && request.auth.uid == deviceId;
    }

    // Devices Collection
    match /devices/{deviceId} {
      // Parent can read/update their child's device. Device can read/update itself.
      allow read, update: if isParentOfDevice(deviceId) || isDeviceItself(deviceId);
      // Device can register itself
      allow create: if isAuthenticated(); 
    }

    // Locations Collection
    match /locations/{locationId} {
      allow read: if isParentOfDevice(resource.data.deviceId);
      allow create: if isDeviceItself(request.resource.data.deviceId);
    }

    // Photos Collection
    match /photos/{photoId} {
      allow read: if isParentOfDevice(resource.data.deviceId);
      allow create: if isDeviceItself(request.resource.data.deviceId);
    }

    // Audio Recordings Collection
    match /audio_recordings/{audioId} {
      allow read: if isParentOfDevice(resource.data.deviceId);
      allow create: if isDeviceItself(request.resource.data.deviceId);
    }

    // SMS Logs Collection
    match /sms_logs/{smsId} {
      allow read: if isParentOfDevice(resource.data.deviceId);
      allow create: if isDeviceItself(request.resource.data.deviceId);
    }

    // Call Logs Collection
    match /call_logs/{callId} {
      allow read: if isParentOfDevice(resource.data.deviceId);
      allow create: if isDeviceItself(request.resource.data.deviceId);
    }
  }
}
